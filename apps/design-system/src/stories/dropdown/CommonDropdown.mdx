## Usage

<Source code={`
import {
  CommonChip,
  CommonDropdown,
  CommonDropdownInner,
  CommonDropdownItem,
  CommonDropdownItemProps,
} from '@yeong/ui';

export default function Page() {
  const [value, setValue] = useState('')

const list: CommonDropdownItemProps = [
{
children: '테스트1',
value: 'test1',
},
{
children: '테스트2',
value: 'test2',
},
{
children: '테스트3',
value: 'test3',
},
];

const changeValue = (value: string) => {
setValue(value)
}

return (

<CommonDropdown value={value} onChange={changeValue}>
  <CommonDropdownInner>
    {list.map((item) => (
      <CommonDropdownItem value={item.value} key={item.value}>
        {item.children}
      </CommonDropdownItem>
    ))}
  </CommonDropdownInner>
</CommonDropdown>
)

}
`} />
